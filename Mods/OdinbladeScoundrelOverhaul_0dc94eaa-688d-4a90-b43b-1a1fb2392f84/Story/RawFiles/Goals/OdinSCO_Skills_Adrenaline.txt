SubGoalCombiner SGC_AND
INITSECTION

KBSECTION
//Character used the non-Combo version of Adrenaline
IF
CharacterUsedSkill(_Character, "Shout_Adrenaline", _, _)
AND
HasActiveStatus(_Character, "OdinSCO_USE_COMBO_3", 0)
THEN
ApplyStatus(_Character, "OdinSCO_APPLY_ADRENALINE", 6.0);

//Slow the character if they used the non-Combo version of Adrenaline
IF
ObjectTurnEnded(_Character)
AND
HasActiveStatus(_Character, "OdinSCO_APPLY_ADRENALINE", 1)
AND
CharacterIsInCombat((CHARACTERGUID)_Character, 1)
THEN
ApplyStatus(_Character, "SLOWED", 6.0, 1, _Character);
RemoveStatus(_Character, "OdinSCO_APPLY_ADRENALINE");
EXITSECTION

ENDEXITSECTION
